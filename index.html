splits = goog.splits

def adj_split(x):
  split_adj = 1
  trade_date = x["TradeDate"]
  for idx in splits.index:
    if(trade_date < idx.tz_localize(None)):
      split_adj *= splits[idx] 
  return split_adj

def lin_reg(x, y):
  x =	np.array(x).reshape(-1, 1)
  y =	np.array(y).reshape(-1, 1)
  fit = LinearRegression().fit(x, y)
  return fit.coef_[0][0]
